def choose_existing_path(candidates):
    for p in candidates:
        if os.path.exists(p):
            return p
    raise FileNotFoundError(f"No CSV found in candidates: {','.join(candidates)}")

def rmse(y_true, y_pred):
    return float(np.sqrt(mean_squared_error(y_true, y_pred)))

def mae(y_true, y_pred):
    return float(mean_absolute_error(y_true, y_pred))

def safe_r2(y_true, y_pred):
    y_true = np.asarray(y_true)
    y_pred = np.asarray(y_pred)
    if len(y_true) < 2: return np.nan
    if np.nanstd(y_true) == 0: return np.nan
    return float(r2_score(y_true, y_pred))

def clip_nonnegative(x):
    x = np.asarray(x, dtype=float)
    return np.clip(x, 0, None)

def ensure_datetime(df, col='Date'):
    if col not in df.columns:
        raise KeyError(f"Missing required column: {col}")
    df[col] = pd.to_datetime(df[col], errors='coerce')
    return df

def standardize_reflectance_columns(df):
    df = df.copy()
    df.columns = df.columns.str.strip()
    
    rename_map = {
        'rho_rc Blue': 'rhorc_Blue', 'rho_rc_Blue': 'rhorc_Blue',
        'rho_rc Green': 'rhorc_Green', 'rho_rc_Green': 'rhorc_Green',
        'rho_rc Red': 'rhorc_Red', 'rho_rc_Red': 'rhorc_Red',
        'rho_rc SWIR1': 'rhorc_SWIR1', 'rho_rc_SWIR1': 'rhorc_SWIR1',
        'Rrs_Blue': 'Rrs_Blue', 'Rrs_Green': 'Rrs_Green',
        'Rrs_Red': 'Rrs_Red', 'Rrs_SWIR1': 'Rrs_SWIR1',
    }
    existing_map = {k: v for k, v in rename_map.items() if k in df.columns}
    if existing_map:
        df = df.rename(columns=existing_map)
        
    if 'date' in df.columns and 'Date' not in df.columns:
        df = df.rename(columns={'date': 'Date'})
    return df

def find_swir1_column(df):
    normalized = {c: c.lower().replace(' ', '').replace('_', '').replace('.', '').replace('-', '') for c in df.columns}
    swir_candidates = []
    for c, key in normalized.items():
        if ('rhorc' in key and 'swir1' in key) or ('rhorc' in key and '1610' in key) or ('rhorc' in key and '1606' in key):
            swir_candidates.append(c)
    
    if not swir_candidates:
        return None, []
    
    if 'rhorc_SWIR1' in df.columns:
        base_col = 'rhorc_SWIR1'
    else:
        base_col = swir_candidates[0]
    return base_col, swir_candidates

def unify_swir1_column(df):
    df = df.copy()
    base_col, swir_candidates = find_swir1_column(df)
    
    if base_col is None:
        raise KeyError("Cannot find SWIR1 reflectance column (e.g. rhorc_SWIR1 / rhorc_1610 / rhorc_1606).")
        
    series = pd.to_numeric(df[base_col], errors='coerce').copy()
    for c in swir_candidates:
        if c == base_col: continue
        series = series.combine_first(pd.to_numeric(df[c], errors='coerce'))
        
    df['rhorc_SWIR1'] = series
    return df

def collapse_duplicate_columns(df):
    df = df.copy()
    df = df.loc[:, ~df.columns.duplicated()]
    return df

def find_makou_files():
    date_nc = MAKOU_DATE_NC
    discharge_nc = MAKOU_DISCHARGE_NC
    
    if date_nc and os.path.exists(date_nc) and discharge_nc and os.path.exists(discharge_nc):
        return date_nc, discharge_nc

    print(f"Searching for Makou NC files in {MAKOU_SEARCH_DIR}...")
    for root, dirs, files in os.walk(MAKOU_SEARCH_DIR):
        for f in files:
            file_lower = f.lower()
            if f.endswith('.nc') and 'makou' in file_lower:
                full_path = os.path.join(root, f)
                if 'date' in file_lower:
                    date_nc = full_path
                elif 'discharge' in file_lower or 'flow' in file_lower or 'q' in file_lower:
                    discharge_nc = full_path

    if not date_nc or not discharge_nc:
        error_msg = (
            f"Cannot locate Makou NC files automatically in {MAKOU_SEARCH_DIR}.\n"
            f"Please run `find {MAKOU_SEARCH_DIR} -name '*makou*.nc'`\n"
            f"and paste the absolute paths into MAKOU_DATE_NC and MAKOU_DISCHARGE_NC at the top of the script."
        )
        raise FileNotFoundError(error_msg)
        
    print(f"Found Date NC: {date_nc}")
    print(f"Found Discharge NC: {discharge_nc}")
    return date_nc, discharge_nc


def load_makou_qlag():
    print("CSV lacks Qlag columns, attempting to load Makou discharge NC...")
    
    date_nc, discharge_nc = find_makou_files()

    with xr.open_dataset(date_nc) as ds_date:
        datenum_vals = ds_date['datenum'].values
        
    with xr.open_dataset(discharge_nc) as ds_q:
        discharge_vals = ds_q['discharge'].values
        
    dates = pd.to_datetime(datenum_vals - MATLAB_EPOCH_OFFSET, unit='D', origin='unix').normalize()
    
    dfq = pd.DataFrame({'date': dates, 'Q': np.ravel(discharge_vals)}) \
            .sort_values('date') \
            .drop_duplicates('date') \
            .reset_index(drop=True)
            
    full_idx = pd.date_range(dfq['date'].min(), dfq['date'].max(), freq='D')
    dfq = dfq.set_index('date').reindex(full_idx).reset_index()
    dfq.columns = ['date', 'Q']
    
    neg_mask = dfq['Q'] <= 0
    n_neg = int(neg_mask.sum())
    if n_neg > 0:
        dfq.loc[neg_mask, 'Q'] = np.nan
        dfq['Q'] = dfq['Q'].interpolate(method='linear')
        
    dfq['Qlag1'] = dfq['Q'].shift(1)
    dfq['Qlag3'] = dfq['Q'].rolling(3, min_periods=1).mean().shift(1)
    dfq['Qlag7'] = dfq['Q'].rolling(7, min_periods=1).mean().shift(1)
    dfq['logQlag3'] = np.log1p(dfq['Qlag3'])
    
    dfq = dfq.dropna(subset=['Qlag1', 'Qlag3', 'Qlag7']).reset_index(drop=True)
    
    print(f"negative-Q count (fixed): {n_neg}")
    for col in ['Qlag1', 'Qlag3', 'Qlag7']:
        print(f"  {col:8s} min={int(dfq[col].min())} median={int(dfq[col].median())} max={int(dfq[col].max())} m3/s")
    
    valid_days = int(dfq['date'].dt.year.isin(range(2017, 2024)).sum())
    print(f"2017-2023 valid days: {valid_days}")
    return dfq

def merge_qlag_features(df, dfq):
    df = df.copy()
    if 'Date' in df.columns and 'date' not in df.columns:
        df['date'] = pd.to_datetime(df['Date'], errors='coerce').dt.normalize()
    else:
        df['date'] = pd.to_datetime(df['date'], errors='coerce').dt.normalize()
        
    n_before = len(df)
    df = df.merge(
        dfq[['date', 'Qlag1', 'Qlag3', 'Qlag7', 'logQlag3']],
        on='date',
        how='inner'
    )
    month = df['date'].dt.month
    df['month_sin'] = np.sin(2 * np.pi * month / 12.0)
    df['month_cos'] = np.cos(2 * np.pi * month / 12.0)
    df['Year'] = df['date'].dt.year.astype(int)
    
    if 'Date' not in df.columns:
        df['Date'] = df['date']
        
    print(f"Qlag merge: before={n_before}, after={len(df)}, dropped={n_before - len(df)}")
    if len(df) > 0:
        print(f"merged year range: {df['Year'].min()}-{df['Year'].max()}")
    return df

def clean_and_engineer_features(df):
    df = df.copy()
    df.columns = df.columns.str.strip()
    
    for c in df.columns:
        if df.columns.tolist().count(c) > 1:
            df = df.loc[:, ~df.columns.duplicated()]
            
    df = standardize_reflectance_columns(df)
    df = ensure_datetime(df, 'Date')
    df = unify_swir1_column(df)
    
    required_cols = ['Date', 'SS', 'rhorc_Blue', 'rhorc_Green', 'rhorc_Red', 'logQlag3', 'Qlag1', 'Qlag7']
    missing = [c for c in required_cols if c not in df.columns]
    if missing:
        raise KeyError(f"Missing required columns: {missing}")
        
    for col in ['SS', 'rhorc_Blue', 'rhorc_Green', 'rhorc_Red', 'logQlag3', 'Qlag1', 'Qlag7']:
        df[col] = pd.to_numeric(df[col], errors='coerce')
        
    df['month_sin'] = np.sin(2 * np.pi * df['Date'].dt.month / 12.0)
    df['month_cos'] = np.cos(2 * np.pi * df['Date'].dt.month / 12.0)
    
    df['ratioRG'] = df['rhorc_Red'] / (df['rhorc_Green'] + EPS)
    df['ratioRB'] = df['rhorc_Red'] / (df['rhorc_Blue'] + EPS)
    df['ratioGB'] = df['rhorc_Green'] / (df['rhorc_Blue'] + EPS)
    
    # REMOVED ratioRS1G per v18c spec
    features = [
        'rhorc_Blue', 'rhorc_Green', 'rhorc_Red',
        'ratioRG', 'ratioRB', 'ratioGB', 
        'logQlag3', 'Qlag1', 'Qlag7',
        'month_sin', 'month_cos'
    ]
    
    keep_cols = ['Date', 'SS'] + features
    df = df[keep_cols].copy()
    
    df = df.replace([np.inf, -np.inf], np.nan)
    df = df.dropna(subset=keep_cols).copy()
    df = df[df['SS'] > 0].copy()
    df['Year'] = df['Date'].dt.year.astype(int)
    
    df = df.sort_values('Date').reset_index(drop=True)
    
    print(f"len(df)={len(df)} samples, len(features)={len(features)}")
    print(f"Features : {features}")
    print("SS : min={:.2f} median={:.2f} mean={:.2f} max={:.2f} mg/L".format(
        df['SS'].min(), df['SS'].median(), df['SS'].mean(), df['SS'].max()
    ))
    
    return df, features

def build_model(model_name):
    if model_name == 'RandomForest':
        return RandomForestRegressor(
            n_estimators=600, max_depth=None, min_samples_split=4, min_samples_leaf=2,
            max_features='sqrt', n_jobs=-1, random_state=RANDOM_STATE
        )
    elif model_name == 'XGBoost':
        return xgb.XGBRegressor(
            n_estimators=1200, learning_rate=0.03, max_depth=3, min_child_weight=3,
            subsample=0.85, colsample_bytree=0.85, reg_alpha=0.0, reg_lambda=2.0,
            objective='reg:squarederror', random_state=RANDOM_STATE, tree_method='hist',
            early_stopping_rounds=50
        )
    elif model_name == 'LightGBM':
        return lgb.LGBMRegressor(
            n_estimators=1200, learning_rate=0.03, num_leaves=31, max_depth=-1,
            min_child_samples=10, subsample=0.85, colsample_bytree=0.85,
            reg_alpha=0.0, reg_lambda=2.0, objective='regression',
            random_state=RANDOM_STATE, verbose=-1
        )
    elif model_name == 'CatBoost':
        return CatBoostRegressor(
            iterations=1200, learning_rate=0.03, depth=6, l2_leaf_reg=5.0,
            loss_function='RMSE', random_seed=RANDOM_STATE, verbose=False
        )
    elif model_name == 'SVR':
        return Pipeline([
            ('scaler', StandardScaler()),
            ('svr', SVR(C=10.0, epsilon=0.2, gamma='scale', kernel='rbf'))
        ])
    else:
        raise ValueError(f"Unknown model_name: {model_name}")

def fit_with_inner_validation(model_name, X_train, y_train):
    model = build_model(model_name)
    use_inner_val = model_name in ['XGBoost', 'LightGBM', 'CatBoost'] and len(X_train) > 25
    
    if use_inner_val:
        X_sub, X_val, y_sub, y_val = train_test_split(
            X_train, y_train, test_size=0.2, random_state=RANDOM_STATE
        )
        if model_name == 'XGBoost':
            model.fit(X_sub, y_sub, eval_set=[(X_val, y_val)], verbose=False)
        elif model_name == 'LightGBM':
            model.fit(
                X_sub, y_sub, eval_set=[(X_val, y_val)], eval_metric='l2',
                callbacks=[lgb.early_stopping(50, verbose=False)]
            )
        elif model_name == 'CatBoost':
            model.fit(X_sub, y_sub, eval_set=(X_val, y_val), use_best_model=True, verbose=False)
    else:
        model.fit(X_train, y_train)
        
    return model

def fit_full_model(model_name, X_all, y_all):
    return fit_with_inner_validation(model_name, X_all, y_all)

def plot_scatter(y_true, y_pred, years, model_name, out_png):
    y_true = np.asarray(y_true)
    y_pred = np.asarray(y_pred)
    years = np.asarray(years)
    r2 = safe_r2(y_true, y_pred)
    the_rmse = rmse(y_true, y_pred)
    the_mae = mae(y_true, y_pred)
    
    uniq_years = sorted(pd.unique(years))
    cmap = plt.get_cmap('tab10')
    color_map = {yr: cmap(i % 10) for i, yr in enumerate(uniq_years)}
    
    fig, ax = plt.subplots(figsize=(6.8, 6.2))
    for yr in uniq_years:
        m = (years == yr)
        ax.scatter(
            y_true[m], y_pred[m], s=28, alpha=0.72,
            color=color_map[yr], label=str(yr), edgecolor='white', linewidth=0.4
        )
        
    vmax = max(float(np.max(y_true)), float(np.max(y_pred))) * 1.05
    vmax = max(vmax, 10)
    
    ax.plot([0, vmax], [0, vmax], '--', color='black', lw=1.2)
    ax.set_xlim(0, vmax)
    ax.set_ylim(0, vmax)
    ax.set_xlabel('Observed SS (mg/L)')
    ax.set_ylabel('Predicted SS (mg/L)')
    ax.set_title(f"{model_name} LOYO Scatter ({VERSION})")
    
    text = f"R² = {r2:.3f}\nRMSE = {the_rmse:.2f} mg/L\nMAE = {the_mae:.2f} mg/L\nN = {len(y_true)}"
    ax.text(
        0.04, 0.96, text, transform=ax.transAxes, va='top', ha='left',
        fontsize=10, bbox=dict(boxstyle='round', fc='white', ec='0.7', alpha=0.9)
    )
    
    ax.grid(alpha=0.25)
    ax.legend(loc='lower right', fontsize=8, ncol=2, frameon=True)
    plt.tight_layout()
    plt.savefig(out_png, bbox_inches='tight')
    plt.close(fig)

def plot_model_comparison(stats_df, out_png):
    dfp = stats_df.sort_values('R2_LOYO', ascending=False).reset_index(drop=True)
    fig, axes = plt.subplots(1, 2, figsize=(12, 4.8))
    
    axes[0].bar(dfp['Model'], dfp['R2_LOYO'], color='#4C78A8')
    axes[0].set_title(f"LOYO R² ({VERSION})")
    axes[0].set_ylabel('R²')
    axes[0].grid(axis='y', alpha=0.25)
    for i, v in enumerate(dfp['R2_LOYO']):
        axes[0].text(i, v + 0.01, f"{v:.3f}", ha='center', va='bottom', fontsize=9)
        
    axes[1].bar(dfp['Model'], dfp['RMSE_LOYO'], color='#F58518')
    axes[1].set_title(f"LOYO RMSE ({VERSION})")
    axes[1].set_ylabel('RMSE (mg/L)')
    axes[1].grid(axis='y', alpha=0.25)
    for i, v in enumerate(dfp['RMSE_LOYO']):
        axes[1].text(i, v + 0.05, f"{v:.2f}", ha='center', va='bottom', fontsize=9)
        
    for ax in axes:
        ax.tick_params(axis='x', rotation=20)
        
    plt.tight_layout()
    plt.savefig(out_png, bbox_inches='tight')
    plt.close(fig)

def run_loyo_all_models(df, features):
    years = sorted(df['Year'].unique().tolist())
    
    print("-" * 60)
    print(f"{VERSION.upper()} LOYO-CV (No ratioRS1G)")
    print("-" * 60)
    
    print(f"{VERSION.upper()} LOYO fold counts:")
    counts = df['Year'].value_counts().sort_index()
    max_count = counts.max()
    for yr, n in counts.items():
        bar = "#" * max(1, int(round(n / max_count * 15)))
        print(f"  {yr}: {n:3d} {bar}")
    print(f"  (fold count min={counts.min()} max={counts.max()})\n")
    
    model_stats = []
    all_pred_records = []
    full_models = {}
    
    X_all = df[features].copy()
    y_all = df[TARGET].copy()
    
    for model_name in MODEL_ORDER:
        print(f"[{model_name}] ({len(years)} folds)...")
        t_model_start = time.time()
        
        fold_rows = []
        y_true_all, y_pred_all, year_all = [], [], []
        
        for i, test_year in enumerate(years, start=1):
            fold_start = time.time()
            
            train_df = df[df['Year'] != test_year].copy()
            test_df = df[df['Year'] == test_year].copy()
            
            X_train = train_df[features].values
            y_train = train_df[TARGET].values
            X_test = test_df[features].values
            y_test = test_df[TARGET].values
            
            model = fit_with_inner_validation(model_name, X_train, y_train)
            y_pred = clip_nonnegative(model.predict(X_test))
            
            fold_r2 = safe_r2(y_test, y_pred)
            fold_rmse = rmse(y_test, y_pred)
            fold_mae = mae(y_test, y_pred)
            fold_time = time.time() - fold_start
            
            print(f"  fold {i:2d}/{len(years)} (year {test_year}, n={len(test_df):3d}): R2={fold_r2:6.3f} RMSE={fold_rmse:5.2f} MAE={fold_mae:5.2f} ({fold_time:4.1f}s)")
            
            fold_rows.append({
                'Model': model_name, 'Year': int(test_year), 'n_test': int(len(test_df)),
                'R2': fold_r2, 'RMSE': fold_rmse, 'MAE': fold_mae, 'FoldTime_s': fold_time
            })
            
            pred_fold = test_df[['Date', 'Year', 'SS']].copy()
            pred_fold['Model'] = model_name
            pred_fold['Pred_SS'] = y_pred
            pred_fold['AbsErr'] = np.abs(pred_fold['Pred_SS'] - pred_fold['SS'])
            all_pred_records.append(pred_fold)
            
            y_true_all.extend(y_test.tolist())
            y_pred_all.extend(y_pred.tolist())
            year_all.extend([test_year] * len(test_df))
            
        fold_df = pd.DataFrame(fold_rows)
        
        full_model = fit_full_model(model_name, X_all.values, y_all.values)
        full_models[model_name] = deepcopy(full_model)
        
        train_pred = clip_nonnegative(full_model.predict(X_all.values))
        train_r2 = safe_r2(y_all.values, train_pred)
        
        loyo_r2 = safe_r2(y_true_all, y_pred_all)
        loyo_rmse = rmse(y_true_all, y_pred_all)
        loyo_mae = mae(y_true_all, y_pred_all)
        overfit_gap = train_r2 - loyo_r2
        total_time = time.time() - t_model_start
        
        scatter_png = os.path.join(OUTPUT_DIR, f"scatter_{model_name.lower()}_{VERSION}_loyo.png")
        plot_scatter(
            y_true=np.array(y_true_all), y_pred=np.array(y_pred_all),
            years=np.array(year_all), model_name=model_name, out_png=scatter_png
        )
        
        model_stats.append({
            'Model': model_name, 'R2_Train': train_r2, 'R2_LOYO': loyo_r2,
            'Overfit_Gap': overfit_gap, 'RMSE_LOYO': loyo_rmse, 'MAE_LOYO': loyo_mae,
            'TrainTime_s': total_time
        })
        
        print(f"  -> {model_name} Train R²={train_r2:.3f} | LOYO R²={loyo_r2:.3f} (Gap={overfit_gap:.3f}) | RMSE={loyo_rmse:.2f} MAE={loyo_mae:.2f} | total={total_time:.0f}s\n")
        
        fold_csv = os.path.join(OUTPUT_DIR, f"fold_metrics_{model_name.lower()}_{VERSION}.csv")
        fold_df.to_csv(fold_csv, index=False)
        
    stats_df = pd.DataFrame(model_stats).sort_values(['R2_LOYO', 'RMSE_LOYO'], ascending=[False, True]).reset_index(drop=True)
    pred_df = pd.concat(all_pred_records, axis=0).reset_index(drop=True)
